cols = 3
rows = 6
marker_size = 5.
marker_spacing = 7.

board_width  = (marker_spacing)*(cols-1) + marker_size
board_height = (marker_spacing)*(rows-1) + marker_size

pts = makeMarkerGrid(rows, cols, marker_size, width = board_width, height=board_height)

pdf = makeBundlePdf([pts], marker_size=marker_size, start_index = 0, text_size = .25)
xml_str = makeBundleXml([pts], marker_size=marker_size, start_index = 0)

pdf.output('example_bundles/test_bundle.pdf', 'F')
with open('example_bundles/test_bundle.xml', 'w') as f:
    f.write(xml_str)
